.contenido.activo {
  max-height: 200px;
  padding: 10px;

const botones = document.querySelectorAll(".boton");
const contenidos = document.querySelectorAll(".contenido");

botones.forEach((boton, index) => {
  boton.addEventListener("click", () => {
    const estaActivo = contenidos[index].classList.contains("activo");

    contenidos.forEach((contenido, i) => {
      contenido.classList.remove("activo");
      botones[i].textContent = "Mostrar sección " + (i + 1);
    });

    if (!estaActivo) {
      contenidos[index].classList.add("activo");
      boton.textContent = "Ocultar sección " + (index + 1);
    }
  });
});
